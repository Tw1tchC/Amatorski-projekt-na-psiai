using System; using System.Collections.Generic; class Program { static List SitoEratostenesa(int n) { bool[] A = new bool[n + 1]; // krok 1: ustaw wszystkie wartości na true for (int i = 2; i <= n; i++) { A[i] = true; } // krok 2: wykreślanie wielokrotności for (int p = 2; p * p <= n; p++) { if (A[p]) { for (int i = p * p; i <= n; i += p) { A[i] = false; } } } // krok 3: zbieranie liczb pierwszych List primes = new List(); for (int i = 2; i <= n; i++) { if (A[i]) primes.Add(i); } return primes; } static void Main() { Console.Write("Podaj n (2-100): "); int n; if (int.TryParse(Console.ReadLine(), out n) && n >= 2 && n <= 100) { List liczbyPierwsze = SitoEratostenesa(n); Console.WriteLine("Liczby pierwsze do " + n + ":"); Console.WriteLine(string.Join(", ", liczbyPierwsze)); } else { Console.WriteLine("Błąd: n musi być w zakresie 2-100"); } } }
